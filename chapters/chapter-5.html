Chapter 5

Advanced React Patterns

1. Higher-Order Components (HOC)

HOC adalah fungsi yang menerima komponen dan mengembalikan komponen baru dengan kemampuan tambahan. Pattern ini banyak dipakai sebelum hooks muncul, tapi masih relevan untuk cross-cutting concerns.

// HOC: withAuth
function withAuth(Component) {
  return function AuthenticatedComponent(props) {
    const { user, isLoading } = useAuth();

    if (isLoading) return <Spinner />;
    if (!user) return <Navigate to='/login' />;

    return <Component {...props} user={user} />;
  };
}

// Pakai HOC
const ProtectedDashboard = withAuth(Dashboard);
const ProtectedSettings = withAuth(Settings);

// HOC: withLoading
function withLoading(Component, fetchDataFn) {
  return function EnhancedComponent(props) {
    const [data, setData] = useState(null);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState(null);

    useEffect(() => {
      fetchDataFn()
        .then(d => setData(d))
        .catch(e => setError(e))
        .finally(() => setLoading(false));
    }, []);

    if (loading) return <Spinner />;
    if (error) return <Error message={error.message} />;
    return <Component data={data} {...props} />;
  };
}

💡 Tip: HOC itu seperti jaket — kamu pakai jaket (HOC) di atas baju (component). Jaket memberikan fungsi tambahan (auth, loading, tracking) tanpa mengubah baju di dalamnya.

2. Render Props

Render props pattern memberikan kontrol penuh ke parent tentang bagaimana child harus dirender. Daripada hardcode UI di dalam child, child menerima fungsi sebagai prop yang mengembalikan JSX.

// Render Props: MouseTracker
function MouseTracker({ render }) {
  const [position, setPosition] = useState({ x: 0, y: 0 });

  useEffect(() => {
    const handler = (e) => setPosition({ x: e.clientX, y: e.clientY });
    window.addEventListener('mousemove', handler);
    return () => window.removeEventListener('mousemove', handler);
  }, []);

  return render(position);
}

// Parent yang tentukan rendering
function App() {
  return (
    <MouseTracker render={({ x, y }) => (
      <h1>Mouse: {x}, {y}</h1>
    )} />
  );
}

// Versi children prop (lebih umum)
function DataFetcher({ url, children }) {
  const [data, setData] = useState(null);
  useEffect(() => { fetch(url).then(r => r.json()).then(setData); }, [url]);
  return children(data);
}

<DataFetcher url='/api/users'>
  {(users) => users.map(u => <div key={u.id}>{u.name}</div>)}
</DataFetcher>

💡 Tip: Render props itu seperti restoran — kamu pesan bahan mentah (data), tapi kamu sendiri yang masak (render) sesuai selera. Lebih fleksibel daripada komponen yang sudah fix tampilannya.

3. Compound Components

Compound components memungkinkan sekumpulan komponen bekerja sama untuk menyelesaikan satu tugas. Parent mengatur state, child mengakses state via context. Contoh klasik: select/option, tabs, accordion.

import { createContext, useContext, useState } from 'react';

// 1. Context untuk share state
const TabsContext = createContext();

// 2. Parent: manage state
function Tabs({ children, defaultValue }) {
  const [activeTab, setActiveTab] = useState(defaultValue);
  return (
    <TabsContext.Provider value={{ activeTab, setActiveTab }}>
      <div>{children}</div>
    </TabsContext.Provider>
  );
}

// 3. Child: akses shared state
function TabList({ children }) {
  return <div className='flex gap-2'>{children}</div>;
}

function Tab({ value, children }) {
  const { activeTab, setActiveTab } = useContext(TabsContext);
  const isActive = activeTab === value;
  return (
    <button
      onClick={() => setActiveTab(value)}
      className={isActive ? 'bg-accent text-dark-900' : 'bg-dark-700 text-gray-300'}
    >{children}</button>
  );
}

function TabPanel({ value, children }) {
  const { activeTab } = useContext(TabsContext);
  if (activeTab !== value) return null;
  return <div>{children}</div>;
}

// Pakai
function App() {
  return (
    <Tabs defaultValue='tab1'>
      <TabList>
        <Tab value='tab1'>Profil</Tab>
        <Tab value='tab2'>Settings</Tab>
      </TabList>
      <TabPanel value='tab1'><ProfilForm /></TabPanel>
      <TabPanel value='tab2'><SettingsForm /></TabPanel>
    </Tabs>
  );
}

💡 Tip: Compound components itu seperti set LEGO — setiap blok punya fungsi sendiri, tapi kalau dirangkai jadi satu, membentuk sesuatu yang utuh. API-nya natural: <Tabs><Tab /><TabPanel /></Tabs>.

4. Container vs Presentational Pattern

Memisahkan komponen yang mengurus logic (container) dari yang mengurus tampilan (presentational). Ini membuat kode lebih terorganisir dan reusable.

// Presentational: fokus tampilan saja, data dari props
function UserList({ users, onSelect, isLoading }) {
  if (isLoading) return <div>Loading...</div>;
  return (
    <ul>
      {users.map(user => (
        <li key={user.id} onClick={() => onSelect(user.id)}>
          {user.name} - {user.email}
        </li>
      ))}
    </ul>
  );
}

// Container: fokus logic, data fetching
function UserListContainer() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/users')
      .then(r => r.json())
      .then(data => { setUsers(data); setLoading(false); });
  }, []);

  const handleSelect = (id) => {
    console.log('Selected:', id);
  };

  return (
    <UserList
      users={users}
      onSelect={handleSelect}
      isLoading={loading}
    />
  );
}

💡 Tip: Container = chef di dapur (urus resep & bahan). Presentational = pelayan di ruang makan (hanya sajikan). Saat ini pattern ini sudah banyak digantikan custom hooks, tapi konsep pemisahan logic-UI tetap relevan.
